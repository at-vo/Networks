Key * key_construct(char *in_name, int in_id)

BStree bstree_ini(int size)
{
	BStree_struct *bsts;
    	BStree bst;
    	Node *nodes;
    	unsigned char *chars;
	nodes = (Node *)calloc(size + 1, sizeof(Node));
    	chars = (unsigned char *)calloc(size + 1, sizeof(unsigned char));
    	bsts->tree_nodes = nodes;
    	bsts->is_free = chars;
    	bsts->size = size;
	bst = bsts;
	return bst;
}
